       public class InspectionViewModel : Observable {
           private readonly InspectionRecipeManager _recipeManager;
           private ObservableCollection<FAIResultRow> _faiResults;

           public ObservableCollection<FAIResultRow> FAIResults {
               get { return _faiResults; }
               set { _faiResults = value; RaisePropertyChanged("FAIResults"); }
           }

           public InspectionViewModel(InspectionRecipeManager recipeManager) {
               _recipeManager = recipeManager;
               _faiResults = new ObservableCollection<FAIResultRow>();
           }

           /// <summary>Called when a FAI node is selected in InspectionListView. Updates result table.</summary>
           public void OnFAISelected(FAIConfig fai) {
               var rows = new ObservableCollection<FAIResultRow>();
               // Add the FAI's own measurement as a row
               rows.Add(new FAIResultRow(fai));
               // Add grouped measurements if any
               foreach (var m in fai.Measurements) {
                   // MeasurementItem results will be shown in Phase 3
               }
               FAIResults = rows;
           }

           /// <summary>Called when an Action node is selected. Shows all FAIs under that action.</summary>
           public void OnActionSelected(NodeViewModel actionNode) {
               var rows = new ObservableCollection<FAIResultRow>();
               if (actionNode.Children != null) {
                   foreach (var child in actionNode.Children) {
                       if (child.NodeType == ENodeType.FAI && child.Param is FAIConfig fai) {
                           rows.Add(new FAIResultRow(fai));
                       }
                   }
               }
               FAIResults = rows;
           }

           public void ClearResults() {
               FAIResults = new ObservableCollection<FAIResultRow>();
           }

           /// <summary>Adds a new FAI to the recipe manager.</summary>
           public FAIConfig AddFAI(string name) {
               return _recipeManager.AddFAI(_recipeManager, name);
           }

           /// <summary>Removes a FAI by index from the recipe manager.</summary>
           public bool RemoveFAI(int index) {
               return _recipeManager.RemoveFAI(index);
           }
       }
       ```

       private void Btn_AddFAI_Click(object sender, RoutedEventArgs e) {
           try {
           // Get currently selected action node
           if (!(treeListBox_sequence.SelectedItem is NodeViewModel selectedNode)) return;
           NodeViewModel actionNode = selectedNode;
           if (selectedNode.NodeType == ENodeType.FAI) {
               actionNode = selectedNode.Parent;
           }
           if (actionNode == null || actionNode.NodeType != ENodeType.Action) return;

           string defaultName = "FAI_" + (actionNode.Children?.Count ?? 0);
           if (!TextInputBox.Show("FAI 이름 입력", defaultName, out string name)) return;

           FAIConfig newFai = _inspectionVm.AddFAI(name);
           if (newFai != null) {
               ViewModel.AddFAINode(actionNode, newFai, actionNode.SequenceID, actionNode.ActionID);
           }
           } catch (Exception ex) { CustomMessageBox.Show("FAI 추가 오류", ex.Message); }
       }

       private void Btn_RemoveFAI_Click(object sender, RoutedEventArgs e) {
           try {
           if (!(treeListBox_sequence.SelectedItem is NodeViewModel selectedNode)) return;
           if (selectedNode.NodeType != ENodeType.FAI) return;
           if (!(selectedNode.Param is FAIConfig fai)) return;

           // Per D-10: confirmation dialog
           MessageBoxResult result = CustomMessageBox.ShowConfirmation(
               "FAI 삭제",
               $"FAI \"{fai.FAIName}\"을(를) 삭제합니다. 계속하시겠습니까?",
               MessageBoxButton.YesNo);
           if (result != MessageBoxResult.Yes) return;

           // Find index in recipe manager
           int index = SystemHandler.Handle.Sequences.RecipeManager.FAIs.IndexOf(fai);
           if (index >= 0) {
               _inspectionVm.RemoveFAI(index);
               selectedNode.Detach();
           }
           } catch (Exception ex) { CustomMessageBox.Show("FAI 삭제 오류", ex.Message); }
       }

       private void Btn_RenameFAI_Click(object sender, RoutedEventArgs e) {
           try {
           if (!(treeListBox_sequence.SelectedItem is NodeViewModel selectedNode)) return;
           if (selectedNode.NodeType != ENodeType.FAI) return;
           if (!(selectedNode.Param is FAIConfig fai)) return;

           if (!TextInputBox.Show("FAI 이름 수정", fai.FAIName, out string newName)) return;
           fai.FAIName = newName;
           selectedNode.Name = newName;
           } catch (Exception ex) { CustomMessageBox.Show("FAI 이름 수정 오류", ex.Message); }
       }
       ```
