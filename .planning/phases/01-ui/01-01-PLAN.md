       ```csharp
       public bool SaveFAIs(IniFile saveFile) {
           saveFile["FAIS"]["Count"] = FAIs.Count;
           for (int i = 0; i < FAIs.Count; i++) {
               string section = $"FAI_{i}";
               FAIConfig fai = FAIs[i];
               fai.Save(saveFile, section);
               saveFile[section]["FAIName"] = fai.FAIName ?? $"FAI_{i}";
               saveFile[section]["ZPosition"] = fai.ZPosition;
               saveFile[section]["DelayMs"] = fai.DelayMs;
               saveFile[section]["SimulImagePath"] = fai.SimulImagePath ?? "";
           }
           return true;
       }

       public bool LoadFAIs(IniFile loadFile) {
           if (!loadFile.ContainsSection("FAIS")) return false;
           int count = loadFile["FAIS"]["Count"].ToInt();
           ClearFAIs();
           for (int i = 0; i < count; i++) {
               string section = $"FAI_{i}";
               if (!loadFile.ContainsSection(section)) continue;
               FAIConfig fai = AddFAI(_owner);
               fai.Load(loadFile, section);
               fai.FAIName = loadFile[section]["FAIName"].ToString();
               fai.ZPosition = loadFile[section]["ZPosition"].ToDouble();
               fai.DelayMs = loadFile[section]["DelayMs"].ToInt();
               fai.SimulImagePath = loadFile[section]["SimulImagePath"].ToString();
           }
           return true;
       }

       public bool HasFAIFormatData(IniFile iniFile) {
           return iniFile.ContainsSection("FAIS");
       }
       ```
