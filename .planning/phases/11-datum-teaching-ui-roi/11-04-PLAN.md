       ```csharp
       //260423 hbk Phase 11 D-20 — 노드 상태 배지: Datum/FAI/Measurement 상태 → Red/Yellow/Green/Transparent
       public System.Windows.Media.Brush StatusColor
       {
           get
           {
               switch (this.NodeType)
               {
                   case ENodeType.Datum:
                   {
                       var d = this.Param as ReringProject.Sequence.DatumConfig;
                       if (d == null) return _badgeTransparent;
                       return d.IsConfigured ? (System.Windows.Media.Brush)_badgeGreen : _badgeRed;
                   }
                   case ENodeType.FAI:
                   {
                       var f = this.Param as ReringProject.Sequence.FAIConfig;
                       if (f == null) return _badgeTransparent;
                       // D-20 Circle-only FAI: use Circle_Radius > 0 instead of ROI_Length*
                       bool circleOnly = IsCircleOnlyFai(f);
                       bool roiSet;
                       if (circleOnly) {
                           var circle = FirstCircleMeasurement(f);
                           roiSet = (circle != null) && (circle.Circle_Radius > 0);
                       } else {
                           roiSet = (f.ROI_Length1 > 0) && (f.ROI_Length2 > 0);
                       }
                       return roiSet ? (System.Windows.Media.Brush)_badgeGreen : _badgeRed;
                   }
                   case ENodeType.Measurement:
                   {
                       var m = this.Param as ReringProject.Sequence.MeasurementBase;
                       if (m == null) return _badgeTransparent;
                       if (m.TolerancePlus == 0 && m.ToleranceMinus == 0) return _badgeYellow;
                       return _badgeGreen;
                   }
                   default:
                       return _badgeTransparent;
               }
           }
       }

       private static bool IsCircleOnlyFai(ReringProject.Sequence.FAIConfig f)
       {
           if (f.MeasurementList == null || f.MeasurementList.Count == 0) return false;
           foreach (var m in f.MeasurementList)
               if (!(m is ReringProject.Sequence.CircleDiameterMeasurement)) return false;
           return true;
       }

       private static ReringProject.Sequence.CircleDiameterMeasurement FirstCircleMeasurement(ReringProject.Sequence.FAIConfig f)
       {
           if (f.MeasurementList == null) return null;
           foreach (var m in f.MeasurementList) {
               var c = m as ReringProject.Sequence.CircleDiameterMeasurement;
               if (c != null) return c;
           }
           return null;
       }
       ```
