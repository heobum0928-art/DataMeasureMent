#### B. D-03 Pattern ROI 동시 렌더링 접근

```csharp
//260517 hbk Phase 23.1 D-03 — EdgeToLineDistanceMeasurement Point ROI 추가 수집
private List<RoiDefinition> GetCurrentFAIRois() {
    var result = new List<RoiDefinition>();
    var seen = new HashSet<string>();
    foreach (var item in dataGrid_faiResults.Items) {
        var row = item as MeasurementResultRow;
        if (row == null || string.IsNullOrEmpty(row.FAIName)) continue;
        if (!seen.Add(row.FAIName)) continue;
        FAIConfig fai = FindFAIByName(row.FAIName);
        if (fai == null) continue;
        var roi = fai.ToRoiDefinition();
        if (roi.IsTaught) result.Add(roi);

        //260517 hbk Phase 23.1 D-03 — Measurement Point ROI 추가 (EdgeToLineDistance)
        foreach (var m in fai.Measurements) {
            var etl = m as EdgeToLineDistanceMeasurement;
            if (etl != null && etl.Point_Length1 > 0 && etl.Point_Length2 > 0) {
                result.Add(new RoiDefinition {
                    Id = fai.FAIName + "_" + (string.IsNullOrEmpty(etl.MeasurementName) ? etl.TypeName : etl.MeasurementName),
                    Name = etl.MeasurementName ?? etl.TypeName,
                    Row1 = etl.Point_Row - etl.Point_Length1,
                    Column1 = etl.Point_Col - etl.Point_Length2,
                    Row2 = etl.Point_Row + etl.Point_Length1,
                    Column2 = etl.Point_Col + etl.Point_Length2,
                    IsTaught = true
                });
            }
        }
    }
    return result;
}
```
