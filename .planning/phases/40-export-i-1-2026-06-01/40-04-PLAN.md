using (var wb = new XLWorkbook())
{
    var ws = wb.Worksheets.Add("Result");
    // D-06 메타 헤더 블록 (행 1~3)
    ws.Cell(1,1).Value = "모델명";   ws.Cell(1,2).Value = cycle.RecipeName;
    ws.Cell(2,1).Value = "검사일시"; ws.Cell(2,2).Value = cycle.InspectionTime.ToString("yyyy-MM-dd HH:mm:ss");
    ws.Cell(3,1).Value = "종합판정"; ws.Cell(3,2).Value = cycle.OverallJudgement;
    // D-05 테이블 헤더 (행 5)
    int hr = 5;
    string[] headers = { "Shot","FAI","측정명","Nominal","Tol+","Tol-","측정값","판정","이미지" };
    for (int i = 0; i < headers.Length; i++) ws.Cell(hr, i+1).Value = headers[i];
    int row = hr + 1;
    foreach (var shot in cycle.Shots ?? new List<ShotResultDto>())
        foreach (var fai in shot.FAIs ?? new List<FaiResultDto>())
            foreach (var m in fai.Measurements ?? new List<MeasurementResultDto>())
            {
                ws.Cell(row,1).Value = shot.ShotName;
                ws.Cell(row,2).Value = fai.FAIName;
                ws.Cell(row,3).Value = m.MeasurementName;
                ws.Cell(row,4).Value = m.NominalValue;
                ws.Cell(row,5).Value = m.TolerancePlus;
                ws.Cell(row,6).Value = m.ToleranceMinus;
                if (m.LastHasResult) ws.Cell(row,7).Value = m.LastMeasuredValue; else ws.Cell(row,7).Value = "-";
                ws.Cell(row,8).Value = m.LastSkipReason == "DATUM_FAIL" ? "DETECT FAIL"
                                       : (m.LastHasResult ? (m.LastJudgement ? "OK" : "NG") : "-");
                // D-07 하이퍼링크 (security T-40-11: 내부 생성 경로만, 절대 경로 정규화)
                if (!string.IsNullOrEmpty(shot.ResultImagePath) && File.Exists(shot.ResultImagePath))
                {
                    string abs = Path.GetFullPath(shot.ResultImagePath);   // 정규화 (T-40-11)
                    ws.Cell(row,9).Value = "이미지 열기";
                    ws.Cell(row,9).Hyperlink = new XLHyperlink(abs);
                }
                row++;
            }
    wb.SaveAs(outputPath);
    return true;
}
catch (Exception ex) { Logging.PrintErrLog((int)ELogType.Error, "[ExcelExportService] Export failed: " + ex.Message); return false; }
```

private void Button_ExportExcel_Click(object sender, RoutedEventArgs e)
{
    if (_currentCycle == null) { CustomMessageBox.Show("엑셀 export", "먼저 cycle 을 선택하세요.", ...); return; }
    // 저장 위치: 해당 cycle 폴더 기본 + 사용자 지정 (D-10)
    var dlg = new Microsoft.Win32.SaveFileDialog {
        Filter = "Excel 파일 (*.xlsx)|*.xlsx",
        FileName = "result_" + _currentCycle.InspectionTime.ToString("yyyyMMdd_HHmmss") + ".xlsx",
        InitialDirectory = (!string.IsNullOrEmpty(_currentCycle.CycleFolderPath) && System.IO.Directory.Exists(_currentCycle.CycleFolderPath))
                           ? _currentCycle.CycleFolderPath : SystemHandler.Handle.Setting.ResultSavePath
    };
    if (dlg.ShowDialog() == true)
    {
        bool ok = ExcelExportService.Export(_currentCycle, dlg.FileName);
        CustomMessageBox.Show("엑셀 export", ok ? "저장 완료: " + dlg.FileName : "export 실패 (로그 확인)", ...);
    }
}
```
