```csharp
public override bool Open(params object[] param) {
    try {
        MIL.MappAlloc(MIL.M_DEFAULT, ref MilApplication);
        if (MilApplication == MIL.M_NULL) throw new Exception("MappAlloc failed");
#if SIMUL_MODE
        MIL.MsysAlloc(MIL.M_DEFAULT, MIL.M_SYSTEM_HOST, MIL.M_DEFAULT, MIL.M_DEFAULT, ref MilSystem);
#else
```

**GrabHalconImage()** — 핵심 (RESEARCH Pattern 2 + PATTERNS GrabHalconImage). SIMUL 폴백 + MdigGrab + M_LOCK/M_HOST_ADDRESS/M_PITCH_BYTE + 명시적 IntPtr 변환(Pitfall 3) + pitch 분기 + 회전 + lock(Interlock):
```csharp
public override HImage GrabHalconImage() {
#if SIMUL_MODE
    return LastHalconImage;   // base GetCurrentImageNoLock() 파일 grab 폴백 (D-11)
#else
    try {
        MIL.MdigGrab(MilDigitizer, MilBuffer);

        MIL_INT hostPtr   = MIL.M_NULL;
        MIL_INT pitchByte = MIL.M_NULL;
        MIL.MbufControl(MilBuffer, MIL.M_LOCK, MIL.M_DEFAULT);
        MIL.MbufInquire(MilBuffer, MIL.M_HOST_ADDRESS, ref hostPtr);
        MIL.MbufInquire(MilBuffer, MIL.M_PITCH_BYTE,   ref pitchByte);

        if (hostPtr == MIL.M_NULL) {
            MIL.MbufControl(MilBuffer, MIL.M_UNLOCK, MIL.M_DEFAULT);
            return null;
        }

        IntPtr ptr = new IntPtr((long)hostPtr);   // Pitfall 3: 명시적 변환
        HImage sourceImage = new HImage();
        if (pitchByte == Info.Width) {
            sourceImage.GenImage1("byte", Info.Width, Info.Height, ptr);   // HikCamera L455-456 동일
        }
        else {
            sourceImage = CreateImageFromPaddedBuffer(ptr, Info.Width, Info.Height, (int)pitchByte);
        }
        MIL.MbufControl(MilBuffer, MIL.M_UNLOCK, MIL.M_DEFAULT);

        HImage rotatedImage = sourceImage;
        if (Info.RotateAngle == ERotateAngleType._90)  { rotatedImage = sourceImage.RotateImage(90.0, "constant");  sourceImage.Dispose(); }
        else if (Info.RotateAngle == ERotateAngleType._180) { rotatedImage = sourceImage.RotateImage(180.0, "constant"); sourceImage.Dispose(); }
        else if (Info.RotateAngle == ERotateAngleType._270) { rotatedImage = sourceImage.RotateImage(270.0, "constant"); sourceImage.Dispose(); }

        lock (Interlock) {
            if (LastGrabHalconImage != null) LastGrabHalconImage.Dispose();
            LastGrabHalconImage = rotatedImage;
        }
        Interlocked.Increment(ref imageCount);
        return LastHalconImage;
    }
    catch (Exception e) {
        Logging.PrintLog((int)ELogType.Camera, "[ERROR] {0} MilCamera.GrabHalconImage ({1})", Name, e.Message);
        Interlocked.Increment(ref errorCount);
        return null;
    }
#endif
}
```
주의: `LastGrabHalconImage` 가 base 에서 protected 접근 가능한지 read_first 로 확인. 만약 private/없음이면 HikCamera 가 어떻게 설정하는지 확인 후 동일 멤버명/접근으로 처리 (Rule 1).
`?.Dispose()` (null-conditional) 대신 명시적 `if (x != null)` 사용 — Phase 20 D-04 스타일 정합 + C# 7.2 안전.

**CreateImageFromPaddedBuffer 헬퍼** — pitch > width padding 행 복사 (RESEARCH Pitfall 4, unsafe 블록 허용):
```csharp
private HImage CreateImageFromPaddedBuffer(IntPtr src, int width, int height, int pitchByte) {
    // pitch padding 제거하여 연속 버퍼로 복사 후 GenImage1
    byte[] packed = new byte[width * height];
    unsafe {
        byte* s = (byte*)src;
        for (int row = 0; row < height; row++) {
            Marshal.Copy((IntPtr)(s + row * pitchByte), packed, row * width, width);
        }
    }
    HImage img = new HImage();
    GCHandle h = GCHandle.Alloc(packed, GCHandleType.Pinned);
    try {
        img.GenImage1("byte", width, height, h.AddrOfPinnedObject());
        // GenImage1 가 복사하지 않을 수 있으므로 즉시 독립 복사본 반환
        HImage copy = img.CopyImage();
        img.Dispose();
        return copy;
    }
    finally {
        h.Free();
    }
}
```
