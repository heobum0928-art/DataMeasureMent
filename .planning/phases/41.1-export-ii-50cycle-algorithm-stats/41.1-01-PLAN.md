```csharp
//260612 hbk Phase 41.1 OUT-03 50회 반복 실행 서비스
public class RepeatRunService
{
    public const int DEFAULT_REPEAT_COUNT = 50;

    // 완료 시 발화: arg = 누적된 모든 CycleResultDto 리스트
    public event Action<List<CycleResultDto>> OnRepeatComplete;

    // 진행 상황 통지: (현재 횟수, 목표 횟수)
    public event Action<int, int> OnProgressChanged;

    public bool IsRunning { get; private set; }
    public int CompletedCount { get; private set; }
    public int TargetCount { get; private set; }

    private InspectionSequence _seq;
    private List<CycleResultDto> _collected;
    private EventSequenceStateChanged _onFinishHandler;
    private object _lock = new object();

    // seq: 반복할 InspectionSequence (UI 에서 전달)
    // targetCount: 반복 횟수 (기본 DEFAULT_REPEAT_COUNT)
    public void Start(InspectionSequence seq, int targetCount = DEFAULT_REPEAT_COUNT)
    public void Stop()  // 강제 중단 (이벤트 구독 해제)
}
```

Start(seq, targetCount) 구현:
1. IsRunning=true 이면 return (중복 시작 방지)
2. _seq = seq, TargetCount = targetCount, CompletedCount = 0, _collected = new List
3. _onFinishHandler = (ctx) => HandleFinish(ctx) 람다 저장
4. seq.OnFinish += _onFinishHandler
5. TriggerNext() 호출 → seq.StartAll(null) [D-08: RequestPacket=null = 수동 경로]

HandleFinish(SequenceContext ctx) — OnFinish 핸들러:
1. lock(_lock) 내에서 진행
2. var recipeManager = SystemHandler.Handle.Sequences.RecipeManager
3. var result = seq.ComputeOverallResult 는 private — 대신 dto.OverallJudgement 기준으로 EVisionResultType 역매핑:
   - "DETECT_FAIL" → EVisionResultType.NotExist
   - "NG" → EVisionResultType.NG
   - 그 외 → EVisionResultType.OK
4. CycleResultDto dto = CycleResultSerializer.BuildDto(recipeManager, resultType, DateTime.Now, SystemHandler.Handle.Setting.CurrentRecipeName, _seq.Name) 동기 호출
5. CycleResultSerializer.SaveAsync(dto) — 기존 경로 영속화 유지 (D-02)
6. _collected.Add(dto)
7. CompletedCount++
8. OnProgressChanged?.Invoke(CompletedCount, TargetCount) — Dispatcher 로 UI 쓰레드 전환 필요 없음 (UI 가 구독하여 Dispatcher 처리)
9. if (CompletedCount >= TargetCount): Stop() 호출 후 OnRepeatComplete?.Invoke(new List<CycleResultDto>(_collected)) 발화
10. else: TriggerNext() 호출

TriggerNext():
- seq.State == EContextState.Idle 이면 seq.StartAll(null)
- 아니면 짧은 지연 후 재시도 → System.Threading.Tasks.Task.Delay(50).ContinueWith(_ => TriggerNext())

Stop():
1. if (_seq != null && _onFinishHandler != null): _seq.OnFinish -= _onFinishHandler
2. IsRunning = false
3. _onFinishHandler = null
4. _seq = null

필요한 using:
