```csharp
private void ApplyCheckerboardCalibration(CalibrationResult result) {
    if (result == null) return;
    double mmPerPixel = result.MmPerPixel;
    string activeSeq = ResolveActiveSequenceForCalibration();

    string warnLine = result.IsDistortionWarn
        ? string.Format("\n[경고] 외곽 왜곡 {0:F2}% — undistort 검토 권장", result.CenterOuterDeviationPct) : "";
    string msg = string.Format("활성 시퀀스 [{0}] 전체 SHOT 의 PixelResolution 을\n1 px = {1:F5} mm 로 덮어씁니다.{2}\n적용하시겠습니까?",
        activeSeq, mmPerPixel, warnLine);
    if (CustomMessageBox.Show(msg, "캘리브레이션 적용", /* Yes/No */ ) != /* Yes */) return;   // D-06 확인 게이트

    InspectionRecipeManager recipeManager = SystemHandler.Handle.Sequences.RecipeManager;
    if (recipeManager == null) { CustomMessageBox.Show("레시피 매니저를 찾을 수 없습니다.", "캘리브레이션"); return; }
    int applied = 0;
    for (int i = 0; i < recipeManager.ShotCount; i++) {
        ShotConfig shot = recipeManager.Shots[i];
        string owner = string.IsNullOrEmpty(shot.OwnerSequenceName) ? SequenceHandler.SEQ_TOP : shot.OwnerSequenceName;
        if (owner != activeSeq) continue;
        shot.PixelResolution = mmPerPixel;                       // Phase 42 단일소스
        foreach (FAIConfig fai in shot.FAIList) {
            fai.PixelResolutionX = mmPerPixel;                   // INI 호환 보존
            fai.PixelResolutionY = mmPerPixel;
        }
        applied++;
    }

    // 저장 — REUSE MainWindow.SaveRecipe (existingFile 보존, 비활성 시퀀스 소실 가드 / Pitfall 4)
    MainWindow mw = Window.GetWindow(this) as MainWindow;
    if (mw != null) mw.SaveRecipe();
    CustomMessageBox.Show(string.Format("{0}개 SHOT 에 적용 + 저장 완료 (1 px = {1:F5} mm)", applied, mmPerPixel), "캘리브레이션");
}
```
