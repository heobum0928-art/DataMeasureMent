```csharp
//260618 hbk Phase 54 ALIGN-01 패턴매칭 보정 합성 (D-02 ②③④). 매칭 x,y → 이동 line-fit θ → rigid → _datumTransforms 합성.
//  refImage = 보정 전 원본 grab 이미지(D-05). modelPath = RecipeFileHelper.GetPatternModelFilePath 결과(호출부 전달).
//  실패(매칭 score 미달/모델 로드 실패/line-fit 실패) → false (호출부가 MarkDatumFailed, lenient D-10). _datumTransforms 미변경.
public bool TryComposeAlign(DatumConfig datum, HImage refImage, string modelPath, out string error)
{
    error = null;
    if (datum == null) { error = "datum null"; return false; }
    var svc = new ReringProject.Halcon.Algorithms.PatternMatchService();
    double curRow, curCol, curAngleDeg, curScore;
    // ① 매칭 (보정 전 원본) → x,y
    if (!svc.TryFindPose(refImage, datum.PatternEngine, modelPath,
            datum.PatternRoi_Row, datum.PatternRoi_Col, datum.PatternRoi_Length1, datum.PatternRoi_Length2,
            datum.PatternSearchMarginPx, datum.PatternMinScore, /*downsampleFactor*/ 2.0,
            out curRow, out curCol, out curAngleDeg, out curScore, out error))
    {
        return false; // ALIGN_FAIL 신호는 호출부에서 MarkDatumFailed
    }
    // ② x,y 로 line-fit ROI 이동 후 정밀 θ. (이동량 dRow=curRow-RefMatchRow, dCol=curCol-RefMatchCol)
    //    DatumFindingService.TryGetLevelingAngle 재사용 — 이동된 ROI 좌표로 line-fit (planner 재량: ROI offset 인자 or 임시 datum 복제).
    double thetaRad = 0.0;
    var dfs = new ReringProject.Halcon.Algorithms.DatumFindingService();
    string thErr;
    if (!dfs.TryGetLevelingAngle(refImage, datum, out thetaRad, out thErr))
    {
        thetaRad = 0.0; // line-fit 실패 → θ=0 폴백 (x,y 보정은 유지, lenient)
    }
    // ③ rigid 산출
    HTuple alignRigid;
    if (!svc.TryBuildAlignRigid(datum.RefMatchRow, datum.RefMatchCol, datum.RefMatchAngleDeg,
            curRow, curCol, thetaRad, out alignRigid, out error))
    {
        return false;
    }
    // ④ 기존 검출 transform(있으면)과 합성 — 없으면 alignRigid 단독 저장
    string datumKey = datum.DatumName; if (datumKey == null) datumKey = "";
    HTuple existing;
    HTuple composed;
    if (_datumTransforms.TryGetValue(datumKey, out existing) && existing != null && existing.Length > 0)
    {
        try { HOperatorSet.HomMat2dCompose(alignRigid, existing, out composed); }
        catch (Exception ex) { error = ex.Message; return false; }
    }
    else
    {
        composed = alignRigid;
    }
    _datumTransforms[datumKey] = composed;
    datum.LastFindSucceeded = true;
    return true;
}
```
