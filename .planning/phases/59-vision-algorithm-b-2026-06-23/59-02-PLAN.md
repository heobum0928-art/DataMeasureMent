                modeFileName = "Bottom";

            else {
                modeFileName = "Tray";
            }

        public AlignResult Run(HImage img, EEthernetVisionMode mode) {
            if (img == null || mode == EEthernetVisionMode.None) {
                AlignResult notFound = new AlignResult();
                notFound.Found = false;

                    noRef.Found = false;
                    return noRef;
