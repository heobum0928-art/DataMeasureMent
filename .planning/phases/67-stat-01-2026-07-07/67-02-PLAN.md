RepeatMeasurementStats 재사용 계약 (WPF_Example/Custom/Sequence/Inspection/RepeatMeasurementStats.cs — 확인됨, 수정 금지):
```csharp
class MeasurementStat {   // ComputeAll() 반환 Dictionary 값
    string ShotName; string FAIName; string MeasurementName; string TypeName;
    int N; double Mean; double StdDev; double Range; double Cpk;
    double NominalValue; double TolerancePlus; double ToleranceMinus;
    int OkCount; int NgCount; int DetectFailCount;
}
class RepeatMeasurementStats {
    void AddSample(CycleResultDto dto);              // Shot→FAI→Measurement 3중 루프 누적
    Dictionary<string, MeasurementStat> ComputeAll();// key = ShotName + "/" + FAIName + "/" + MeasurementName
}
// AddSample 정책 (재현 대상):
//   LastSkipReason=="DATUM_FAIL"||"NO_IMAGE" → DetectFailCount++ (값 제외)
//   else if LastHasResult → Values.Add(LastMeasuredValue); LastJudgement ? OkCount++ : NgCount++
//   LastHasResult==false && skip 없음 → 무시(통계 미포함)
// ComputeAll: n>0 mean, n>1 stddev, Cpk=min((usl-mean)/3σ,(mean-lsl)/3σ), σ=0→PositiveInfinity
//   usl=Nominal+TolPlus, lsl=Nominal-|TolMinus|
```

CycleResultDto 재구성 대상 (WPF_Example/UI/ViewModel/CycleResultDto.cs — 확인됨):
```csharp
CycleResultDto { DateTime InspectionTime; string RecipeName; int IndexNumber; string OverallJudgement; List<ShotResultDto> Shots; }

FaiResultDto  { string FAIName; List<MeasurementResultDto> Measurements; }
MeasurementResultDto { string MeasurementName; string TypeName; double NominalValue; double TolerancePlus;
```

    **(H) private static List<string> ParseCsvLine(string szLine) — RFC4180 파서 (따옴표/콤마/`""`):**
    ```csharp
    var fields = new List<string>();
    var sb = new StringBuilder();
    bool bInQuotes = false;
    int i = 0;
    while (i < szLine.Length)
    {
        char c = szLine[i];
        if (bInQuotes)
        {
            if (c == '"')
            {
                if (i + 1 < szLine.Length && szLine[i + 1] == '"') { sb.Append('"'); i += 2; continue; }   // "" → "
                bInQuotes = false; i++; continue;
            }
            sb.Append(c); i++; continue;
        }
        if (c == '"') { bInQuotes = true; i++; continue; }
        if (c == ',') { fields.Add(sb.ToString()); sb.Clear(); i++; continue; }
        sb.Append(c); i++;
    }
    fields.Add(sb.ToString());
    return fields;
    ```
