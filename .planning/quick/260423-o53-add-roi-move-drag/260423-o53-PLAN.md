//260423 hbk ROI hit-test: 클릭 좌표가 선택된 ROI 내부인지
private RoiDefinition HitTestSelectedRoi(Point imagePoint) {
    if (string.IsNullOrEmpty(_selectedRoiId)) return null;
    var roi = _rois.FirstOrDefault(r => r.Id == _selectedRoiId);
    if (roi == null) return null;
    if (roi.Shape == RoiShape.Circle) {
        double dr = imagePoint.Y - roi.CenterRow;
        double dc = imagePoint.X - roi.CenterCol;
        if (Math.Sqrt(dr * dr + dc * dc) <= roi.Radius) return roi;
        return null;
    }
    // Rect bbox
    if (imagePoint.Y >= roi.Row1 && imagePoint.Y <= roi.Row2 &&
        imagePoint.X >= roi.Column1 && imagePoint.X <= roi.Column2) return roi;
    return null;
}

//260423 hbk 드로잉 모드 여부 (이동 차단)
private bool IsAnyDrawingModeActive() {
    return _isDrawingRect || _isDrawingCircle || _polygonDraftPoints != null || _calibrationPoints != null;
}

//260423 hbk ROI 이동 완료 → 모델 반영
private void HalconViewer_RoiMoveCompleted(object sender, RoiMoveCompletedArgs e) {
    if (e == null || string.IsNullOrEmpty(e.RoiId)) return;

    // Circle 대상 먼저 확인 (CircleDiameterMeasurement)
    var faiForCircle = FindFAIByName(e.RoiId);
    if (faiForCircle != null) {
        foreach (var m in faiForCircle.Measurements) {
            if (m is CircleDiameterMeasurement circle) {
                circle.Circle_Row += e.DeltaRow;
                circle.Circle_Col += e.DeltaCol;
                var rois = GetCurrentFAIRois();
                halconViewer.UpdateDisplayState(rois, e.RoiId, null, null);
                return;
            }
        }
        // Rect FAI
        faiForCircle.ROI_Row += e.DeltaRow;
        faiForCircle.ROI_Col += e.DeltaCol;
        var rois2 = GetCurrentFAIRois();
        halconViewer.UpdateDisplayState(rois2, e.RoiId, null, null);
        return;
    }

    // CircleDiameterMeasurement이 별도 id로 저장되는 경우 대비
    var selectedRow = dataGrid_faiResults.SelectedItem as MeasurementResultRow;
    if (selectedRow != null) {
        var fai = FindFAIByName(selectedRow.FAIName);
        if (fai != null) {
            foreach (var m in fai.Measurements) {
                if (m is CircleDiameterMeasurement cm && cm.MeasurementName == e.RoiId) {
                    cm.Circle_Row += e.DeltaRow;
                    cm.Circle_Col += e.DeltaCol;
                    var rois3 = GetCurrentFAIRois();
                    halconViewer.UpdateDisplayState(rois3, e.RoiId, null, null);
                    return;
                }
            }
        }
    }
}
